SELECT user_id, full_name FROM `User`;
SELECT room_type_id, name, hotel_id FROM RoomType;
SELECT stay_date, qty
FROM RoomInventory
WHERE room_type_id = 2
  AND stay_date >= CURRENT_DATE
  AND stay_date < DATE_ADD(CURRENT_DATE, INTERVAL 5 DAY)
ORDER BY stay_date;

  'CARD'                             -- payment method
);
